<%*
// Кладём отчёт в папку «отчёты за день» и именуем по дате создания.
// Перемещаем файл напрямую через API Obsidian, минуя tp.file.move:
// его внутренняя проверка папки ломается на кириллице (NFC/NFD) и
// падает с «Folder already exists».
const folder = "отчёты за день";
// Создаём папку, если её ещё нет. Если уже есть — ошибку игнорируем.
try { await app.vault.createFolder(folder); } catch (e) { /* папка уже есть */ }
// Имя по дате; при конфликте добавляем время.
let fileName = tp.date.now("DD.MM.YYYY");
let newPath = `${folder}/${fileName}.md`;
if (await app.vault.adapter.exists(newPath)) {
    newPath = `${folder}/${fileName} ${tp.date.now("HH-mm")}.md`;
}
await app.fileManager.renameFile(tp.config.target_file, newPath);
-%>
