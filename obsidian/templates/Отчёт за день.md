<%*
// Кладём отчёт в папку «отчёты за день» и именуем по дате создания.
const folder = "отчёты за день";
// Создаём папку, если её ещё нет (Templater сам вложенные папки не создаёт).
if (!app.vault.getAbstractFileByPath(folder)) {
    await app.vault.createFolder(folder);
}
// Если отчёт на сегодня уже есть — добавляем время, чтобы не было конфликта.
let fileName = tp.date.now("DD.MM.YYYY");
if (app.vault.getAbstractFileByPath(`${folder}/${fileName}.md`)) {
    fileName += " " + tp.date.now("HH-mm");
}
await tp.file.move(`${folder}/${fileName}`);
-%>
