// его внутренняя проверка папки ломается на кириллице (NFC/NFD) и
// падает с «Folder already exists».
const folder = "инвестиции/дневник";

let fileName = tp.date.now("DD.MM.YYYY");
let newPath = `${folder}/${fileName}.md`;
if (await app.vault.adapter.exists(newPath)) {
    newPath = `${folder}/${fileName} ${tp.date.now("HH-mm")}.md`;
}
await app.fileManager.renameFile(tp.config.target_file, newPath);
